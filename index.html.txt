const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

let myChart;
const voteCounts = {
 'Pizza': 0,
 'Fast Food': 0,
 'Home Cooked': 0
};

// Initialize Chart
function initChart() {
 const ctx = document.getElementById('voteChart').getContext('2d');
 myChart = new Chart(ctx, {
   type: 'bar',
   data: {
     labels: ['Pizza', 'Fast Food', 'Home Cooked'],
     datasets: [{
       label: 'Votes',
       data: [0, 0, 0],
       backgroundColor: ['#f97316', '#fbbf24', '#34d399']
     }]
   },
   options: {
     responsive: true,
     scales: {
       y: {
         beginAtZero: true,
         ticks: { precision: 0 }
       }
     }
   }
 });
}

// Vote function
async function castVote(choice) {
 const { error } = await supabaseClient
   .from('votes')
   .insert([{ choice: choice }]);

 if (error) {
   console.error(error);
   document.getElementById('status').innerText = "Error submitting vote!";
 } else {
   document.getElementById('status').innerText = " Vote counted!";
 }
}

// Load + realtime
async function setup() {
 initChart();

 // Load existing votes
 const { data, error } = await supabaseClient
   .from('votes')
   .select('choice');

 if (!error && data) {
   data.forEach(v => {
     if (voteCounts[v.choice] !== undefined) {
       voteCounts[v.choice]++;
     }
   });
 }

 updateChart();

 // Realtime updates
 supabaseClient
   .channel('live-votes')
   .on('postgres_changes',
     { event: 'INSERT', schema: 'public', table: 'votes' },
     payload => {
       const choice = payload.new.choice;
       if (voteCounts[choice] !== undefined) {
         voteCounts[choice]++;
         updateChart();
       }
     }
   )
   .subscribe();

 document.getElementById('status').innerText = " Live and connected!";
}

// Update chart
function updateChart() {
 myChart.data.datasets[0].data = [
   voteCounts['Pizza'],
   voteCounts['Fast Food'],
   voteCounts['Home Cooked']
 ];
 myChart.update();
}

setup();
